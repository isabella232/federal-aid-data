# Functions
def classify(string):
    string = re.sub(r'\(|\)', '', string)
    classified = string.lower().strip().replace(' ','-')
    return classified

def find_share(num,total):
    return round((num / total)*100, 2)

def removeSpecChars(string):
    string = string.replace('$','')
    string = string.replace(',','')
    num = float(string)
    return num
